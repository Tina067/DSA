Approach 1:
class Solution {
  public:
    // Function returns the second
    // largest elements
    int getSecondLargest(vector<int> &arr) {
        int n = arr.size();
        if(n==1)
        return -1;
        sort(arr.begin(), arr.end());
        int i = n-2;
        int largest = arr[n-1];
        while(i>=0){
            if(largest != arr[i]){
                return arr[i];
            }
        }
        return -1;
    }
};

Time Complexity: (nlogn)
space Complexity: O(1);
